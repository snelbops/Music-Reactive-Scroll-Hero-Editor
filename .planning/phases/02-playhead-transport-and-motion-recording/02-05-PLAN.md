<!-- Studio scrub API (for smooth drag): -->
```typescript
import studio from '@theatre/studio';
const scrub = studio.scrub();
// During drag:
scrub.capture(({ set }) => {
  sheet.sequence.position = newTimeSeconds;  // set Theatre.js time to target

});
// On release:

// Then refresh dots:
```

<!-- Note: studio.scrub() batches changes into an undoable group. -->
<!-- This is preferable to studio.transaction for drag operations because it -->

In `Timeline.tsx`, add draggable behaviour to the keyframe dot circles in the Scroll POS lane SVG:

1. Add a `draggingKf` ref: `const draggingKfRef = useRef<{ scrub: ReturnType<typeof studio.scrub>; value: number; svgEl: SVGSVGElement } | null>(null)`.

2. Import `studio` from `'@theatre/studio'` (already imported in Plan 04's RecordMode — add here too for Timeline).

3. Change the `<circle>` elements to have pointer event handlers. The SVG uses `preserveAspectRatio="none"` so pointer coordinates need conversion to viewBox coordinates:

```tsx
{keyframeDots.map((kf, i) => (
  <circle
    key={kf.id ?? i}
    cx={(kf.position / SEQUENCE_DURATION) * VB_W}
    cy={(1 - kf.value) * VB_H}
    r="6"
    fill="#a855f7"
    stroke="white"
    strokeWidth="1.5"
    className="cursor-ew-resize"
    style={{ pointerEvents: 'all' }}
    onPointerDown={(e) => {
      e.stopPropagation(); // prevent lane playhead drag from triggering
      (e.target as SVGCircleElement).setPointerCapture(e.pointerId);
      const svgEl = (e.target as SVGCircleElement).ownerSVGElement!;
      draggingKfRef.current = {
        scrub: studio.scrub(),
        value: kf.value,
        svgEl,
      };
    }}
    onPointerMove={(e) => {
      if (!draggingKfRef.current || !(e.buttons & 1)) return;
      const { scrub, value, svgEl } = draggingKfRef.current;
      // Convert client X to SVG viewBox X
      const rect = svgEl.getBoundingClientRect();
      const svgX = ((e.clientX - rect.left) / rect.width) * VB_W;
      const newTime = Math.max(0, Math.min(SEQUENCE_DURATION, (svgX / VB_W) * SEQUENCE_DURATION));
      scrub.capture(({ set }) => {
        sheet.sequence.position = newTime;
        set(scrollControlsObj.props.position, value);
      });
    }}
    onPointerUp={() => {
      if (!draggingKfRef.current) return;
      draggingKfRef.current.scrub.commit();
      draggingKfRef.current = null;
      // Refresh dots after commit
      setKeyframeDots(
        sheet.sequence.__experimental_getKeyframes(scrollControlsObj.props.position) as any[]
      );
    }}
  />
))}
```
