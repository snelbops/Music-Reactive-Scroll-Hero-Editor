<!-- Pointer capture pattern (from research): -->
```typescript
const trackRef = useRef<HTMLDivElement>(null);
const onPointerDown = (e: React.PointerEvent<HTMLDivElement>) => {
  e.currentTarget.setPointerCapture(e.pointerId);
};
const onPointerMove = (e: React.PointerEvent<HTMLDivElement>) => {
  if (!(e.buttons & 1)) return;
  const rect = trackRef.current!.getBoundingClientRect();
  const p = Math.max(0, Math.min(1, (e.clientY - rect.top) / rect.height));
  sheet.sequence.position = p * SEQUENCE_DURATION;
  setSceneProgress(p);
};
// onPointerUp: no action needed — position already snapped on last move
```
