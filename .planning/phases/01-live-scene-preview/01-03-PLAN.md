Add store reads inside the Layout function:


Add an Escape key handler effect inside the Layout function:
```typescript
useEffect(() => {
  const handleKey = (e: KeyboardEvent) => {
    if (e.key === 'Escape' && isFullscreen) setIsFullscreen(false);
  };
  document.addEventListener('keydown', handleKey);
  return () => document.removeEventListener('keydown', handleKey);
}, [isFullscreen, setIsFullscreen]);
```

Add a fullscreen guard before the normal return. When fullscreen is active, render ONLY the Viewport inside a fixed full-window overlay with a small exit button:
```tsx
if (isFullscreen) {

        onClick={() => setIsFullscreen(false)}

        title="Exit fullscreen (Esc)"
      >
        <Minimize2 className="w-4 h-4" />
      </button>
    </div>
  );
}
```
